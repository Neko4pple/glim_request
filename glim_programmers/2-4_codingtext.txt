#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<bool> flag) {
    vector<int> ansr;
    
    for(int i = 0; i < flag.size(); ++i) {
        if(flag[i] == true) {
            for(int j = 0; j < arr[i] * 2; ++j) ansr.push_back(arr[i]);
        }
        else {
            for(int j = 0; j < arr[i]; ++j) ansr.pop_back();
        }
    }
    return ansr;
}
