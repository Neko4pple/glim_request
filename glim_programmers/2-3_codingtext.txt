#include <stdio.h>


int solution(int** dots, size_t dots_rows, size_t dots_cols) {
    int answer=0,a,b; //a는 가로의 길이 b는 세로의 길이
       for(int i = 0; i < dots_rows; i++){
           for(int j = i; j < dots_rows; j++){
               if(dots[i][0] == dots[j][0] && i != j){
                   b = (dots[i][1] - dots[j][1]);   
         
                   if(b < 0){
                       b *= -1;    //길이는 항상 양수로
                   }
               }
               if(dots[i][0] != dots[j][0]){
                   a=(dots[i][0] - dots[j][0]);   
                   if(a<0){
                       a*=-1;     //길이는 항상 양수로
                   }
               }
           }
       }
       
       
    answer = a * b;
    
    return answer;
}
